Java Code Explanation - Player.java

Code:

package Pet;

import java.util.ArrayList;

public class Player {
    private String name;
    private int foodSupply;
    private ArrayList pets;

    public Player(String name) {
        this.name = name;
        this.foodSupply = 50;
        this.pets = new ArrayList<>();
    }

    public void addPet(Pet pet) {
        pets.add(pet);
    }

    public boolean feedPet(int petIndex) {
        if (foodSupply > 0 && petIndex >= 0 && petIndex < pets.size()) {
            Pet pet = pets.get(petIndex);
            pet.feed();
            foodSupply--;
            return true;
        }
        return false;
    }

    public void playWithPet(int petIndex) {
        if (petIndex >= 0 && petIndex < pets.size()) {
            pets.get(petIndex).play();
        }
    }

    public void restPet(int petIndex) {
        if (petIndex >= 0 && petIndex < pets.size()) {
            pets.get(petIndex).rest();
        }
    }

    public String getStatus() {
        StringBuilder status = new StringBuilder("Player: " + name + " | Food: " + foodSupply + " | Pets: " + pets.size() + "\n");
        for (int i = 0; i < pets.size(); i++) {
            status.append(i).append(". ").append(pets.get(i).getStatus()).append("\n");
        }
        return status.toString();
    }

    public void addFood(int foodAmount) {
        foodSupply += foodAmount;
    }

    public int getFoodSupply() {
        return foodSupply;
    }

    public ArrayList getPets() {
        return pets;
    }
}
